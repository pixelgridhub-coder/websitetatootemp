    <script>
        document.addEventListener("DOMContentLoaded", function () {
            const chatButton = document.getElementById("chatButton");
            const chatWindow = document.getElementById("chatWindow");
            const chatClose = document.getElementById("chatClose");
            const chatMessages = document.getElementById("chatMessages");
            const chatInput = document.getElementById("chatInput");
            const chatSend = document.getElementById("chatSend");
            const typingIndicator = document.getElementById("typingIndicator");

            chatButton.onclick = () => chatWindow.classList.toggle("active");
            chatClose.onclick = () => chatWindow.classList.remove("active");

            chatSend.onclick = sendMessage;
            chatInput.addEventListener("keypress", (e) => {
                if (e.key === "Enter") sendMessage();
            });

            function sendMessage() {
                let text = chatInput.value.trim();
                if (!text) return;

                addMessage(text, "user");
                chatInput.value = "";

                typingIndicator.classList.add("active");

                setTimeout(() => {
                    typingIndicator.classList.remove("active");
                    let reply = getBotReply(text);
                    
                    // Check if we need to show appointment button
                    if (reply.hasButton) {
                        addMessageWithButton(reply.text, reply.buttonText);
                    } else {
                        addMessage(reply.text, "bot");
                    }
                }, 900);
            }

            function addMessage(text, type) {
                let msg = document.createElement("div");
                msg.className = `message ${type}-message`;

                msg.innerHTML = `
                    <div class="message-content"><p>${text}</p></div>
                    <div class="message-time">${getTime()}</div>
                `;

                chatMessages.appendChild(msg);
                chatMessages.scrollTop = chatMessages.scrollHeight;
            }

            function addMessageWithButton(text, buttonText) {
                let msg = document.createElement("div");
                msg.className = `message bot-message`;

                msg.innerHTML = `
                    <div class="message-content">
                        <p>${text}</p>
                        <button class="appointment-button" id="appointmentRedirect">${buttonText}</button>
                    </div>
                    <div class="message-time">${getTime()}</div>
                `;

                chatMessages.appendChild(msg);
                
                // Add event listener to the button
                document.getElementById("appointmentRedirect").addEventListener("click", function() {
                    window.location.href = "appointment.html";
                });
                
                chatMessages.scrollTop = chatMessages.scrollHeight;
            }

            function getTime() {
                const t = new Date();
                return t.getHours() + ":" + t.getMinutes().toString().padStart(2, "0");
            }

            /* ---------- AUTO REPLY LOGIC ---------- */
            function getBotReply(msg) {
                msg = msg.toLowerCase();

                // Services related
                if (msg.includes("service") || msg.includes("what do you do") || 
                    msg.includes("tattoo service") || msg.includes("what kind of tattoo")) {
                    return {
                        text: "We provide custom tattoo design, cover-ups, tattoo restoration, and professional consultations. Our artists specialize in various styles from fine-line to traditional.",
                        hasButton: false
                    };
                }

                // Pricing related
                if (msg.includes("price") || msg.includes("cost") || 
                    msg.includes("rate") || msg.includes("charge") || 
                    msg.includes("fee") || msg.includes("how much")) {
                    return {
                        text: "Our pricing depends on design size and complexity. Small tattoos start at â‚¹999. For custom designs, we provide quotes after consultation.",
                        hasButton: false
                    };
                }

                // Hours related
                if (msg.includes("hour") || msg.includes("time") || 
                    msg.includes("open") || msg.includes("when") || 
                    msg.includes("today")) {
                    return {
                        text: "We are open Monday to Friday from 9:00 AM to 8:00 PM, and Saturday from 10:00 AM to 6:00 PM.",
                        hasButton: false
                    };
                }

                // Location related
                if (msg.includes("location") || msg.includes("address") || 
                    msg.includes("where") || msg.includes("place")) {
                    return {
                        text: "We are located at Abirami complex, bus stop, opposite Indian Bank Colony, Ganesh Nagar, K K Nagar, Tiruchirappalli, Tamil Nadu 620021.",
                        hasButton: false
                    };
                }

                // Contact related
                if (msg.includes("contact") || msg.includes("phone") || 
                    msg.includes("call") || msg.includes("number")) {
                    return {
                        text: "You can contact us at +91 63852 17970 or email us at info@qualitytattoostudio.in.",
                        hasButton: false
                    };
                }

                // Appointment related - Now redirects to appointment page
                if (msg.includes("appointment") || msg.includes("book") || 
                    msg.includes("schedule") || msg.includes("reserve") ||
                    msg.includes("booking")) {
                    return {
                        text: "I'd be happy to help you book an appointment! Please click the button below to visit our appointment booking page where you can select your preferred date and time.",
                        hasButton: true,
                        buttonText: "Book Appointment Now"
                    };
                }

                // Safety/Hygiene related
                if (msg.includes("safe") || msg.includes("hygiene") || 
                    msg.includes("clean") || msg.includes("sterile")) {
                    return {
                        text: "We follow strict hygiene practices with 100% sterilized equipment and single-use needles. Your safety is our top priority.",
                        hasButton: false
                    };
                }

                // Aftercare related
                if (msg.includes("aftercare") || msg.includes("healing") || 
                    msg.includes("care") || msg.includes("maintenance")) {
                    return {
                        text: "Keep the tattoo clean, avoid direct sunlight, and apply the recommended healing cream for 7-14 days. We provide detailed aftercare instructions.",
                        hasButton: false
                    };
                }

                // Pain related
                if (msg.includes("pain") || msg.includes("hurt") || 
                    msg.includes("painful")) {
                    return {
                        text: "Pain varies by body part and individual tolerance, but we ensure a comfortable experience with professional techniques and proper aftercare guidance.",
                        hasButton: false
                    };
                }

                // Design related
                if (msg.includes("design") || msg.includes("custom") || 
                    msg.includes("idea") || msg.includes("create")) {
                    return {
                        text: "Yes! We can create custom designs based on your idea or reference image. Our artists will work with you to bring your vision to life.",
                        hasButton: false
                    };
                }

                // Time related
                if (msg.includes("how long") || msg.includes("time") || 
                    msg.includes("session") || msg.includes("duration")) {
                    return {
                        text: "Small tattoos take 20-40 minutes. Larger ones depend on complexity. We'll provide a time estimate during your consultation.",
                        hasButton: false
                    };
                }

                // Ink quality related
                if (msg.includes("ink") || msg.includes("quality") || 
                    msg.includes("material") || msg.includes("safe ink")) {
                    return {
                        text: "We use premium, skin-safe inks that are internationally certified for quality and safety.",
                        hasButton: false
                    };
                }

                // Aftercare timing related
                if (msg.includes("wash") || msg.includes("swim") || 
                    msg.includes("workout") || msg.includes("shower")) {
                    return {
                        text: "Avoid swimming and intense workouts for 7 days. You can gently wash the tattoo with mild soap after 24 hours.",
                        hasButton: false
                    };
                }

                // Default response
                return {
                    text: "Thank you for your message! Our team will respond shortly. For faster assistance, you can call us at +91 63852 17970.",
                    hasButton: false
                };
            }
        });
